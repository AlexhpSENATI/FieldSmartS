import React, { useContext, useState, useMemo } from "react";
import { EstadisticasContext, EstadisticasProvider } from "../../context/EstadisticasContext";
import { Bar } from "react-chartjs-2";
import {
  Chart as ChartJS,
  Title,
  Tooltip,
  Legend,
  BarElement,
  CategoryScale,
  LinearScale
} from "chart.js";

ChartJS.register(Title, Tooltip, Legend, BarElement, CategoryScale, LinearScale);

const rangos = {
  "10s": 10 * 1000,
  "5m": 5 * 60 * 1000,
  "30m": 30 * 60 * 1000,
  "1h": 60 * 60 * 1000,
  "2h": 2 * 60 * 60 * 1000,
  "6h": 6 * 60 * 60 * 1000,
  "12h": 12 * 60 * 60 * 1000,
  "24h": 24 * 60 * 60 * 1000,
  "3d": 3 * 24 * 60 * 60 * 1000,
  "1w": 7 * 24 * 60 * 60 * 1000,
  "1m": 30 * 24 * 60 * 60 * 1000, // Aprox
  "1y": 365 * 24 * 60 * 60 * 1000 // Aprox
};

const nombresMetricas = {
  humedad: "Humedad del suelo",
  humedadAmbiental: "Humedad ambiental",
  temperatura: "Temperatura"
};

const EstadisticasComponente = () => {
  const { datos } = useContext(EstadisticasContext);
  const [rango, setRango] = useState("actual"); // arranca mostrando √∫ltimo dato
  const [metrica, setMetrica] = useState("humedad"); // usada solo en rangos largos

  if (!datos.length) return <p>Cargando datos...</p>;

  // Filtrado de datos seg√∫n rango
  let datosFiltrados = [];
  if (rango === "actual") {
    datosFiltrados = [datos[datos.length - 1]]; // √∫ltimo dato
  } else {
    const ahora = new Date();
    const limite = ahora.getTime() - rangos[rango];
    datosFiltrados = datos.filter((d) => new Date(d.fechaHora).getTime() >= limite);
  }

  let data;
  let labels;

  if (rango === "actual") {
    // ‚úÖ Mostrar las 3 m√©tricas en el √∫ltimo dato
    const ultimo = datosFiltrados[0];
    labels = ["Actual"];
    data = {
      labels,
      datasets: [
        {
          label: nombresMetricas.humedad,
          data: [ultimo.humedad || 0],
          backgroundColor: "rgba(75, 192, 192, 0.5)"
        },
        {
          label: nombresMetricas.humedadAmbiental,
          data: [ultimo.humedadAmbiental || 0],
          backgroundColor: "rgba(153, 102, 255, 0.5)"
        },
        {
          label: nombresMetricas.temperatura,
          data: [ultimo.temperatura || 0],
          backgroundColor: "rgba(255, 159, 64, 0.5)"
        }
      ]
    };
  } else {
    // ‚úÖ Mostrar solo la m√©trica seleccionada
    labels = datosFiltrados.map((d) => d.fechaHora.slice(11, 19));
    const valores = datosFiltrados.map((d) => d[metrica] || 0);

    data = {
      labels,
      datasets: [
        {
          label: nombresMetricas[metrica],
          data: valores,
          backgroundColor: "rgba(75, 192, 192, 0.5)"
        }
      ]
    };
  }

  return (
    <div className="p-4">
      {/* Selector de rango */}
      <div className="mb-4">
        <label className="mr-2">Rango de tiempo:</label>
        <select value={rango} onChange={(e) => setRango(e.target.value)}>
          <option value="actual">Actual (√∫ltimo dato)</option>
          <option value="10s">√öltimos 10 segundos</option>
          <option value="5m">√öltimos 5 minutos</option>
          <option value="30m">√öltimos 30 minutos</option>
          <option value="1h">√öltima 1 hora</option>
          <option value="2h">√öltimas 2 horas</option>
          <option value="6h">√öltimas 6 horas</option>
          <option value="12h">√öltimas 12 horas</option>
          <option value="24h">√öltimas 24 horas</option>
          <option value="3d">√öltimos 3 d√≠as</option>
          <option value="1w">√öltima semana</option>
          <option value="1m">√öltimo mes</option>
          <option value="1y">√öltimo a√±o</option>
        </select>
      </div>

      {/* Selector de m√©trica solo si NO es "actual" */}
      {rango !== "actual" && (
        <div className="mb-4">
          <label className="mr-2">M√©trica:</label>
          <select value={metrica} onChange={(e) => setMetrica(e.target.value)}>
            <option value="humedad">Humedad del suelo</option>
            <option value="humedadAmbiental">Humedad ambiental</option>
            <option value="temperatura">Temperatura</option>
          </select>
        </div>
      )}

      {/* Gr√°fico */}
      <Bar data={data} />
    </div>
  );
};
